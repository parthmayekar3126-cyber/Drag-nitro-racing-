// Space for Brake } void HandleMotor() { // पिछले पहियों में जान डालना (RWD) backLeft.motorTorque = verticalInput * motorForce; backRight.motorTorque = verticalInput * motorForce; float currentbreakForce = isBreaking ? breakForce : 0f; ApplyBreaking(currentbreakForce); } void ApplyBreaking(float force) { frontLeft.brakeTorque = force; frontRight.brakeTorque = force; backLeft.brakeTorque = force; backRight.brakeTorque = force; } void HandleSteering() { float currentSteerAngle = maxSteerAngle * horizontalInput; frontLeft.steerAngle = currentSteerAngle; frontRight.steerAngle = currentSteerAngle; } public float nitroMultiplier = 2f; public bool nitroActive = false; void HandleNitro() { if (Input.GetKey(KeyCode.LeftShift)) { backLeft.motorTorque *= nitroMultiplier; // कार की तरफ देखना var direction = carTarget.position - transform.position; var rotation = Quaternion.LookRotation(direction, Vector3.up); transform.rotation = Quaternion.Lerp(transform.rotation, rotation, lookSpeed * Time.deltaTime);
